    # display rectangle    
    patch = patches.Rectangle((box[1], box[0]), box[3] - box[1], box[2] - box[0],  alpha=0.3)

    #get coordinates for detected face
    x=int(patch.get_x())
    y=int(patch.get_y())
    w=int(patch.get_width())
    h=int(patch.get_height())
    
    
    face=image[y:y+h, x:x+w]
    print(face.shape)
    img=Image.fromarray(face,'RGB')
    img.save(CONFIG_DATA_DIR+str(i)+'.jpg')

    ax.add_patch(patch)

    # display class index and score
    plt.text(x=box[1] + 10, y=box[2] - 10, s='%d (%0.2f) ' % (classes[i], scores[i]), color='w')
